\section{Linear Models for Regression}


\subsection{}
\label{3.1}
By the definition,
%
\begin{equation}
\tanh a 
= 
\frac{\exp(a) - \exp(- a)}{\exp(a) + \exp(- a)}.
\end{equation}
%
The right hand side can be written as
%
\begin{equation}
\frac{1 - \exp(- 2 a)}{1 + \exp(- 2 a)}
=
\frac{2}{1 + \exp(- 2 a)}
-
1.
\end{equation}
%
By the definition
%
\begin{equation}
\sigma (a)
=
\frac{1}{1 + \exp(- a)},
\end{equation}
%
the right hand side can be written as
%
\begin{equation}
\tanh a
=
2
\sigma (2 a)
-
1.
\end{equation}
%

Let 
%
\begin{equation}
y(x_n, \mathbf{w})
=
w_0
+
\sum_{j = 1}^{M}
w_j
\sigma
\left(
	\frac{x - \mu_j}{s}
\right).
\end{equation}
%
By the result above, the right hand side can be written as
%
\begin{equation}
w_0
+
\sum_{j = 1}^{M}
w_j
\frac{\tanh (\frac{x - \mu_j}{2 s}) + 1}{2}
=
w_0
+
\frac{1}{2}
\sum_{j = 1}^{M}
w_j
+
\frac{1}{2}
\sum_{j = 1}^{M}
w_j
\tanh
\left(
	\frac{x - \mu_j}{2 s}
\right).
\end{equation}
%
Therefore, $y(x_n, \mathbf{w})$ is equivalent to
%
\begin{equation}
y(x_n, \mathbf{u})
=
u_0
+
\sum_{j = 1}^{M}
u_j
\tanh
\left(
	\frac{x - \mu_j}{2 s}
\right),
\end{equation}
%
where
%
\begin{equation}
\begin{aligned}
u_0
&=
w_0
+
\frac{1}{2}
\sum_{j = 1}^{M}
w_j, 
\\
u_j
&=
\frac{w_j}{2},
\quad
j
=
1, \cdots, M.
\end{aligned}
\end{equation}
%

Then, for any vector $\mathbf{v}$ in $N$ dimensions,
%
\begin{equation}
\bm{\Phi}
\left(
	\bm{\Phi} ^ \intercal
	\bm{\Phi}
\right) ^ {- 1}
\bm{\Phi} ^ \intercal
\mathbf{v}
\end{equation}
%
is a projection of $\mathbf{v}$ onto the space spanned by the columns of $\mathbf{\Phi}$?

Additionally, for a vector $\mathbf{t}$ in $N$ dimensions,
%
\begin{equation}
\left(
	\bm{\Phi} ^ \intercal
	\bm{\Phi}
\right) ^ {- 1}
\bm{\Phi} ^ \intercal
\mathbf{t}
\end{equation}
%
is an orthogonal projection of $\mathbf{t}$ onto the space spanned by the columns of $\mathbf{\Phi}$?


\subsection{}
\label{3.3}
Let
%
\begin{equation}
E_D (\mathbf{w})
=
\frac{1}{2}
\sum_{n = 1}^{N}
r_n
\left(
	t_n
	-
	\mathbf{w} ^ \intercal
	\bm{\phi} (\mathbf{x}_n)
\right) ^ 2.
\end{equation}
%
Setting the derivative to zero gives
%
\begin{equation}
\mathbf{0}
=
-
\sum_{n = 1}^{N}
r_n
\bm{\phi} (\mathbf{x}_n)
\left(
	t_n
	-
	\mathbf{w} ^ \intercal
	\bm{\phi} (\mathbf{x}_n)
\right).
\end{equation}
%
The right hand side can be written as
%
\begin{equation}
-
\sum_{n = 1}^{N}
r_n
t_n
\bm{\phi} (\mathbf{x}_n)
+
\left(
	\sum_{n = 1}^{N}
	r_n
	\bm{\phi} (\mathbf{x}_n)
	\bm{\phi} (\mathbf{x}_n) ^ \intercal
\right)
\mathbf{w}.
\end{equation}
%
Therefore, 
%
\begin{equation}
\argmin_{\mathbf{w}}
E_D (\mathbf{w})
=
\left(
	\bm{\Phi}'
	{\bm{\Phi}'} ^ \intercal
\right) ^ {- 1}
{\bm{\Phi}'} ^ \intercal
\mathbf{t}',
\end{equation}
%
where
%
\[
\bm{\Phi}'
=
\begin{bmatrix}
\sqrt{r_1} \bm{\phi} (\mathbf{x}_1) ^ \intercal 
\\
\vdots 
\\
\sqrt{r_N} \bm{\phi} (\mathbf{x}_N) ^ \intercal
\end{bmatrix}
,
\bm{t}'
=
\begin{bmatrix}
\sqrt{r_1} t_1 
\\
\vdots 
\\
\sqrt{r_N} t_N
\end{bmatrix}
.
\]
